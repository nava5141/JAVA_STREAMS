import java.util.*;
import java.util.stream.*;

class Employee {

    private int empId;
    private String empName;
    private String department;
    private int experience;
    private double salary;

    // Constructor
    public Employee(int empId, String empName, String department, int experience, double salary) {
        this.empId = empId;
        this.empName = empName;
        this.department = department;
        this.experience = experience;
        this.salary = salary;
    }

    // Getters
    public int getEmpId() { return empId; }
    public String getEmpName() { return empName; }
    public String getDepartment() { return department; }
    public int getExperience() { return experience; }
    public double getSalary() { return salary; }

    @Override
    public String toString() {
        return empId + " " + empName + " " + department + " " + experience + " " + salary;
    }
}

public class Main
{
	public static void main(String[] args) {
	 List<Employee> employees = Arrays.asList(
            new Employee(1, "Ravi", "IT", 3, 40000),
            new Employee(2, "Anjali", "HR", 5, 50000),
            new Employee(3, "Kiran", "IT", 2, 35000),
            new Employee(4, "Sneha", "Finance", 6, 60000),
            new Employee(5, "Arjun", "IT", 4, 45000),
            new Employee(6, "Pooja", "HR", 1, 30000),
            new Employee(7, "Rahul", "Finance", 7, 70000),
            new Employee(8, "Divya", "IT", 3, 42000)
        );
       
                                                            
                                                            
        //1.Find the max salary and min salary in a  map 

        Map<String,Double> maxMin = employees
                                    .stream()
                                    .map(Employee::getSalary)
                                    .collect(Collectors.teeing(
                                        Collectors.maxBy(Double::compare),
                                        Collectors.minBy(Double::compare),
                                        (max,min)-> Map.of("MAX:",max.orElse(0.0),"MIN:",min.orElse(0.0))
                                        ));
                                      //  System.out.println(maxMin);

        //2.Find the total Salary and Total experience

        Map<String,Double> TotSalAndTotExp = employees
                                                .stream()
                                                .collect(Collectors.teeing(
                                                    Collectors.summingDouble(Employee::getSalary),
                                                    Collectors.summingInt(Employee::getExperience),
                                                    (totSal,totExp)-> Map.of("Total Salary: ",totSal,"Total Exp: ",(double)totExp)
                                                    ));
                                                  //  System.out.println(TotSalAndTotExp);

        //3.count of employees and average experience

        Map<String,Long> countAvgExp = employees
                                        .stream()
                                        .collect(Collectors.teeing(
                                            Collectors.counting(),
                                            Collectors.summingInt(Employee::getExperience),
                                            (count,exp)-> Map.of("Count of employees:",count,"Average Exp:",exp/count)
                                            ));
